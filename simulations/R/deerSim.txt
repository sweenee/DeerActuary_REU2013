# Population Distribution

Z10 <- vector(length = 1000)
for (i in 1:1000) {
  dt <- T/N
  dw <- sqrt(dt) * rnorm(N)
  w <- c(0, cumsum(dw[1:(N - 1)]))
  x0 <- F
  z0 <- F/x0
  a <- alpha ^ 2 - r
  b <- alpha
  g0 <- 1 - r/(0.5 * b ^ 2 - a)
  phi <- exp((a - 0.5 * b ^ 2) * T - b * w[N])
  integral <- 0
  ds <- dt
  s = seq(0, T - ds, ds)
  integral = sum(exp((0.5 * b ^ 2 - a) * s + b * w) * dw)
  Ztrue <- r/(0.5 * b ^ 2 - a) + g0 * phi - b * r/(0.5 * b ^ 2 - a) * phi * integral
  Z10[i] <- Ztrue
}
